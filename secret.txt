#include <stdio.h>
#include <string.h>
#include "encode.h"
#include "decode.h"
#include "types.h"

OperationType check_operation_type(char *argv[])
{
    if (argv[1] != NULL && strcmp(argv[1], "-e") == 0)
    {
        return e_encode;
    }
    else if (argv[1] != NULL && strcmp(argv[1], "-d") == 0)
    {
        return e_decode;
    }
    else
    {
        return e_unsupported;
    }
}

int main(int argc, char *argv[])
{
    if (argc < 2)
    {
        printf("Error: Insufficient arguments.\n");
        printf("Usage: ./a.out -e <.bmp file> <.secret file> [output.bmp]\n");
        printf("Usage: ./a.out -d <.bmp file> [output_file_name_without_extn]\n");
        return 1;
    }

    OperationType op_type = check_operation_type(argv);

    if (op_type == e_encode)
    {
        printf("Selected Operation: Encoding\n");

        EncodeInfo encInfo;

        if (read_and_validate_encode_args(argv, &encInfo) == e_success)
        {
            if (do_encoding(&encInfo) == e_success)
            {
                printf("Encoding completed successfully\n");
            }
            else
            {
                printf("Encoding failed\n");
                return 1;
            }
        }
        else
        {
            printf("ERROR: Invalid arguments for encoding\n");
            printf("Usage: ./a.out -e <.bmp file> <.secret file> [output.bmp]\n");
            return 1;
        }
    }
    else if (op_type == e_decode)
    {
        printf("Selected Operation: Decoding\n");

        DecodeInfo decInfo;

        if (read_and_validate_decode_args(argv, &decInfo) == e_success)
        {
            if (do_decoding(&decInfo) == e_success)
            {
                printf("Decoding completed successfully\n");
            }
            else
            {
                printf("Decoding failed\n");
                return 1;
            }
        }
        else
        {
            printf("ERROR: Invalid arguments for decoding\n");
            printf("Usage: ./a.out -d <.bmp file> [output_file_name_without_extn]\n");
            return 1;
        }
    }
    else
    {
        printf("Error: Unsupported operation.\n");
        printf("For Encoding: ./a.out -e <.bmp file> <.secret file> [output.bmp]\n");
        printf("For Decoding: ./a.out -d <.bmp file> [output_file_name_without_extn]\n");
    }

    return 0;
}
